        // ==================== RENDERING ====================
        function renderFeed() {
            const container = document.getElementById('postsContainer');
            container.innerHTML = '';
            posts.forEach(post => {
                const el = document.createElement('div');
                el.className = 'post';
                el.dataset.id = post.id;
                let commentsHTML = post.comments.map(c => `
                    <div class="comment">
                        <img src="https://via.placeholder.com/30">
                        <div class="comment-text"><span class="comment-author">${c.author}</span> ${c.text}</div>
                    </div>
                `).join('');
                const likeClass = post.likedByCurrentUser ? 'action-btn liked' : 'action-btn';
                el.innerHTML = `
                    <div class="post-header">
                        <img src="https://via.placeholder.com/40">
                        <div><div class="post-author">${post.author}</div><div class="post-time">${post.time}</div></div>
                    </div>
                    <div class="post-content">${post.content}</div>
                    <div class="post-stats">
                        <span><i class="fas fa-thumbs-up"></i> <span class="like-count-${post.id}">${post.likes}</span></span>
                        <span>${post.comments.length} comments</span>
                    </div>
                    <div class="post-actions-footer">
                        <div class="like-btn ${likeClass}" data-id="${post.id}"><i class="fas fa-thumbs-up"></i><span>Like</span></div>
                        <div class="comment-btn action-btn" data-id="${post.id}"><i class="fas fa-comment"></i><span>Comment</span></div>
                        <div class="share-btn action-btn" data-id="${post.id}"><i class="fas fa-share"></i><span>Share</span></div>
                    </div>
                    <div class="comments-section" id="comments-${post.id}">
                        ${commentsHTML}
                        <div class="add-comment">
                            <img src="https://via.placeholder.com/30">
                            <input type="text" class="comment-input" data-id="${post.id}" placeholder="Write a comment...">
                            <button class="comment-submit" data-id="${post.id}">Post</button>
                        </div>
                    </div>
                `;
                container.appendChild(el);
            });
            // Reattach event listeners
            document.querySelectorAll('.like-btn').forEach(btn => btn.addEventListener('click', handleLike));
            document.querySelectorAll('.comment-submit').forEach(btn => btn.addEventListener('click', handleCommentSubmit));
            document.querySelectorAll('.share-btn').forEach(btn => btn.addEventListener('click', e => showAlert(`Share post ${e.currentTarget.dataset.id}`)));
            document.querySelectorAll('.comment-btn').forEach(btn => btn.addEventListener('click', e => {
                document.getElementById(`comments-${e.currentTarget.dataset.id}`).scrollIntoView({behavior:'smooth'});
            }));
        }

        function handleLike(e) {
            const id = parseInt(e.currentTarget.dataset.id);
            const post = posts.find(p => p.id === id);
            if (post) {
                post.likedByCurrentUser = !post.likedByCurrentUser;
                post.likes += post.likedByCurrentUser ? 1 : -1;
                renderFeed();
            }
        }

        function handleCommentSubmit(e) {
            const id = parseInt(e.target.dataset.id);
            const input = document.querySelector(`.comment-input[data-id="${id}"]`);
            const text = input.value.trim();
            if (text) {
                const post = posts.find(p => p.id === id);
                post.comments.push({ author: currentUser, text });
                input.value = '';
                renderFeed();
            }
        }

        function renderProfilePosts() {
            const container = document.getElementById('profilePosts');
            container.innerHTML = '';
            const userPosts = posts.filter(p => p.author === currentUser);
            userPosts.forEach(p => {
                const div = document.createElement('div');
                div.className = 'post';
                div.innerHTML = `<div class="post-content">${p.content}</div><div class="post-time">${p.time}</div>`;
                container.appendChild(div);
            });
        }

            friends.forEach(f => {

                div.addEventListener('click', () => showAlert(`View ${f.name}'s profile`));
                container.appendChild(div);

            // Create post
            document.getElementById('postButton').addEventListener('click', () => {
                const input = document.getElementById('postContentInput');
                const content = input.value.trim();
                if (content) {
                    posts.unshift({
                        id: posts.length+1,
                        author: currentUser,
                        content,
                        time: 'Just now',
                        likes: 0,
                        likedByCurrentUser: false,
                        comments: []
                    });
                    input.value = '';
                    renderFeed();

                }
            });

            document.getElementById('photoVideoBtn').addEventListener('click', () => showAlert('Photo/Video upload simulation'));
            document.getElementById('feelingBtn').addEventListener('click', () => showAlert('Feeling/Activity - coming soon'));

                if (e.key === 'Enter') showAlert(`Search for: ${e.target.value}`);

            // Settings tabs
            document.querySelectorAll('.settings-tab').forEach(tab => {
                tab.addEventListener('click', () => {

                    showAlert(`Chat with ${name} (simulated)`);
